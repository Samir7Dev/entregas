    <script>
        const scheduleForm = document.getElementById('scheduleForm');
        const scheduleTable = document.getElementById('scheduleTable').querySelector('tbody');

        scheduleForm.addEventListener('submit', function(e) {
            e.preventDefault();

            const clientName = document.getElementById('clientName').value;
            const vehicleModel = document.getElementById('vehicleModel').value;
            const deliveryDate = document.getElementById('deliveryDate').value;
            const deliveryTime = document.getElementById('deliveryTime').value;
            const deliveryStatus = document.getElementById('deliveryStatus').value;
            const responsiblePerson = document.getElementById('responsiblePerson').value;

            const newRow = document.createElement('tr');
            newRow.innerHTML = `
                <td>${clientName}</td>
                <td>${vehicleModel}</td>
                <td>${deliveryDate}</td>
                <td>${deliveryTime}</td>
                <td>${deliveryStatus}</td>
                <td>${responsiblePerson}</td>
            `;

            scheduleTable.appendChild(newRow);
            scheduleForm.reset();
        });
    </script>
